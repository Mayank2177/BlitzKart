package main

import (
	"net/http"

	"github.com/gin-gonic/gin"
)

func getDispatch(c *gin.Context) {
    c.IndentedJSON(http.StatusOK,dispatch)
}

func postDispatch(c *gin.Context) {
    var newDispatch Dispatch

 // Call BindJSON to bind the received JSON to
    if err := c.BindJSON(&newDispatch); err != nil {
        return
    }
    // Add the new dispatch to the slice.
    dispatch = append(dispatch, newDispatch)
    c.IndentedJSON(http.StatusCreated, newDispatch)
}

func getDispatchByID(c *gin.Context) {
    id := c.Param("id")

    // Loop over the list of albums, looking for
    // an album whose ID value matches the parameter.
    for _, a := range dispatch {
        if a.ID == id {
            c.IndentedJSON(http.StatusOK, a)
            return
        }
    }
    c.IndentedJSON(http.StatusNotFound, gin.H{"message": "dispatch not found"})
}
